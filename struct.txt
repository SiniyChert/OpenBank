структура проекта, сгенерированная GPT

OpenBank/
│
├── app/                      # Основная директория приложения
│   ├── __init__.py          # Инициализация пакета
│   ├── main.py              # Главный файл для запуска FastAPI
│   ├── api/                 # Директория для API
│   │   ├── __init__.py
│   │   ├── routes.py        # Определение маршрутов FastAPI
│   │   └── dependencies.py   # Зависимости для маршрутов
│   │
│   ├── models/              # Директория для моделей базы данных
│   │   ├── __init__.py
│   │   └── user.py          # Модель пользователя
│   │
│   ├── services/            # Директория для бизнес-логики
│   │   ├── __init__.py
│   │   └── bank_service.py   # Логика работы с банковскими операциями
│   │
│   ├── database/            # Директория для работы с базой данных
│   │   ├── __init__.py
│   │   └── db.py            # Подключение к базе данных MySQL
│   │
│   ├── telegram_bot/        # Директория для Telegram-бота
│   │   ├── __init__.py
│   │   └── bot.py           # Логика работы Telegram-бота
│   │
│   └── utils/               # Утилиты и вспомогательные функции
│       ├── __init__.py
│       └── helpers.py       # Вспомогательные функции
│
├── tests/                   # Директория для тестов
│   ├── __init__.py
│   ├── test_api.py          # Тесты для API
│   └── test_services.py     # Тесты для бизнес-логики
│
├── requirements.txt         # Зависимости проекта
├── README.md                # Документация проекта
└── .env                     # Переменные окружения (например, данные для подключения к БД)
